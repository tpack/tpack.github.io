Class Matcher

表示一个路径匹配器

example

const matcher = new Matcher() matcher.include(".js") matcher.include(".jsx") matcher.exclude("y.js") matcher.test("x.js") // true matcher.test("x.jsx") // true matcher.test("y.js") // false
